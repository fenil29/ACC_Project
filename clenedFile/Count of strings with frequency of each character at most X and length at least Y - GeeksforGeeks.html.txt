Input: arr[] = { ?ab?, ?derdee?, ?erre? }, X = 2, Y = 4Output: 1Explanation: Strings with character frequency at most 2 and&nbsp;length at least 4 is ?erre?. Hence count is 1       Input: arr[] = {?ag?, ?ka?, ?nanana?}, X = 3, Y = 2Output: 3   &nbsp;  Approach: Follow the approach mentioned below to solve the problem:  Traverse the array of string, and for each string follow the steps below. Create a frequency map of characters. Whenever any character has a frequency greater than X, or length less than Y, skip the current string. If no character has frequency more than X, and length at least Y, increment the count of answer. Return the count stored in answer when all the strings are traversed.   C++                          #include &lt;bits/stdc++.h&gt;   using namespace std;   &nbsp;&nbsp;   // Function to check if   // the string has   // frequency of each character   // less than X   bool isValid(string s,  int X)   {   &nbsp;&nbsp;&nbsp;&nbsp; vector&lt; int &gt; freq(26, 0);   &nbsp;&nbsp;   &nbsp;&nbsp;&nbsp;&nbsp; // Loop to check the frequency   &nbsp;&nbsp;&nbsp;&nbsp; // of each character in the string   &nbsp;&nbsp;&nbsp;&nbsp; for ( char c : s) {   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; freq++;   &nbsp;&nbsp;&nbsp;&nbsp; }   &nbsp;&nbsp;   &nbsp;&nbsp;&nbsp;&nbsp; // Loop to check   &nbsp;&nbsp;&nbsp;&nbsp; // if the frequency of all characters   &nbsp;&nbsp;&nbsp;&nbsp; // are at most X   &nbsp;&nbsp;&nbsp;&nbsp; for ( int i = 0; i &lt; 26; i++)   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (freq[i] &gt; X)   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; return false ;   &nbsp;&nbsp;&nbsp;&nbsp; return true ;   }   &nbsp;&nbsp;   // Function to calculate the count of strings   int getCount(vector&lt;string&gt;&amp; arr,  int X,  int Y)   {   &nbsp;&nbsp;&nbsp;&nbsp; int ans = 0;   &nbsp;&nbsp;   &nbsp;&nbsp;&nbsp;&nbsp; // Loop to iterate the string array   &nbsp;&nbsp;&nbsp;&nbsp; for (string st : arr) {   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; if (isValid(st, X) &amp;&amp; st.length() &gt;= Y) {   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; ans++;   &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; }   &nbsp;&nbsp;&nbsp;&nbsp; }   &nbsp;&nbsp;&nbsp;&nbsp; return ans;   }   &nbsp;&nbsp;   // Driver Code   int main()   {   &nbsp;&nbsp;&nbsp;&nbsp; vector&lt;string&gt; arr = {  &quot;ab&quot; ,  &quot;derdee&quot; ,  &quot;erre&quot; };   &nbsp;&nbsp;&nbsp;&nbsp; int X = 2, Y = 4;   &nbsp;&nbsp;   &nbsp;&nbsp;&nbsp;&nbsp; // Function call to get count for arr[]   &nbsp;&nbsp;&nbsp;&nbsp; cout &lt;&lt; getCount(arr, X, Y);   &nbsp;&nbsp;&nbsp;&nbsp; return 0;   }                       Output 1  Time Complexity: O(N*M), where N is the size of the array and M is the size of the longest stringAuxiliary
